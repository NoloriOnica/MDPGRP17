#include "motor_encoder.h"


static int32_t last_count_a = 0;
static int32_t last_count_b = 0;

static volatile float rpm_motor_a = 0.0f;
static volatile float rpm_motor_b = 0.0f;

static volatile int32_t encoder_delta_a = 0;
static volatile int32_t encoder_delta_b = 0;

extern TIM_HandleTypeDef htim2;
extern TIM_HandleTypeDef htim3;
extern TIM_HandleTypeDef htim4;
extern TIM_HandleTypeDef htim9;
extern TIM_HandleTypeDef htim6;



void Encoder_Init(void)
{
    HAL_TIM_Encoder_Start(&htim2, TIM_CHANNEL_ALL);
    __HAL_TIM_SET_COUNTER(&htim2, 0);

    HAL_TIM_Encoder_Start(&htim3, TIM_CHANNEL_ALL);
    __HAL_TIM_SET_COUNTER(&htim3, 0);

    last_count_a = 0;
    last_count_b = 0;
}



void Motor_PWM_Init(void)
{
    HAL_TIM_PWM_Start(&htim4, TIM_CHANNEL_3);
    HAL_TIM_PWM_Start(&htim4, TIM_CHANNEL_4);

    __HAL_TIM_SET_COMPARE(&htim4, TIM_CHANNEL_3, 0);
    __HAL_TIM_SET_COMPARE(&htim4, TIM_CHANNEL_4, 0);

    HAL_TIM_PWM_Start(&htim9, TIM_CHANNEL_1);
    HAL_TIM_PWM_Start(&htim9, TIM_CHANNEL_2);

    __HAL_TIM_SET_COMPARE(&htim9, TIM_CHANNEL_1, 0);
    __HAL_TIM_SET_COMPARE(&htim9, TIM_CHANNEL_2, 0);
}



void Encoder_Sampling_Timer_Init(void)
{
    HAL_TIM_Base_Start_IT(&htim6);
}



void MotorA_Forward(uint16_t speed)
{
    if (speed > MOTOR_PWM_MAX) speed = MOTOR_PWM_MAX;
    __HAL_TIM_SET_COMPARE(&htim4, TIM_CHANNEL_3, 0);
    __HAL_TIM_SET_COMPARE(&htim4, TIM_CHANNEL_4, speed);
}

void MotorA_Reverse(uint16_t speed)
{
    if (speed > MOTOR_PWM_MAX) speed = MOTOR_PWM_MAX;
    __HAL_TIM_SET_COMPARE(&htim4, TIM_CHANNEL_3, speed);
    __HAL_TIM_SET_COMPARE(&htim4, TIM_CHANNEL_4, 0);
}

void MotorA_Stop(void)
{
    __HAL_TIM_SET_COMPARE(&htim4, TIM_CHANNEL_3, 0);
    __HAL_TIM_SET_COMPARE(&htim4, TIM_CHANNEL_4, 0);
}

void MotorA_Brake(void)
{
    __HAL_TIM_SET_COMPARE(&htim4, TIM_CHANNEL_3, MOTOR_PWM_MAX);
    __HAL_TIM_SET_COMPARE(&htim4, TIM_CHANNEL_4, MOTOR_PWM_MAX);
}

void MotorB_Forward(uint16_t speed)
{
    if (speed > MOTOR_PWM_MAX) speed = MOTOR_PWM_MAX;
    __HAL_TIM_SET_COMPARE(&htim9, TIM_CHANNEL_1, 0);
    __HAL_TIM_SET_COMPARE(&htim9, TIM_CHANNEL_2, speed);
}

void MotorB_Reverse(uint16_t speed)
{
    if (speed > MOTOR_PWM_MAX) speed = MOTOR_PWM_MAX;
    __HAL_TIM_SET_COMPARE(&htim9, TIM_CHANNEL_1, speed);
    __HAL_TIM_SET_COMPARE(&htim9, TIM_CHANNEL_2, 0);
}

void MotorB_Stop(void)
{
    __HAL_TIM_SET_COMPARE(&htim9, TIM_CHANNEL_1, 0);
    __HAL_TIM_SET_COMPARE(&htim9, TIM_CHANNEL_2, 0);
}

void MotorB_Brake(void)
{
    __HAL_TIM_SET_COMPARE(&htim9, TIM_CHANNEL_1, MOTOR_PWM_MAX);
    __HAL_TIM_SET_COMPARE(&htim9, TIM_CHANNEL_2, MOTOR_PWM_MAX);
}

void Motor_SetSpeed(Motor_ID_t motor, int16_t speed)
{
    uint16_t abs_speed = (speed >= 0) ? (uint16_t)speed : (uint16_t)(-speed);

    switch (motor)
    {
        case MOTOR_A:
            if (speed > 0)       MotorA_Forward(abs_speed);
            else if (speed < 0)  MotorA_Reverse(abs_speed);
            else                 MotorA_Stop();
            break;

        case MOTOR_B:
            if (speed > 0)       MotorB_Forward(abs_speed);
            else if (speed < 0)  MotorB_Reverse(abs_speed);
            else                 MotorB_Stop();
            break;

        default:
            break;
    }
}



int32_t Encoder_GetCount(Motor_ID_t motor)
{
    switch (motor)
    {
        case MOTOR_A:
            return (int32_t)__HAL_TIM_GET_COUNTER(&htim2);
        case MOTOR_B:
            return (int16_t)__HAL_TIM_GET_COUNTER(&htim3);
        default:
            return 0;
    }
}

float Encoder_GetRPM(Motor_ID_t motor)
{
    switch (motor)
    {
        case MOTOR_A:  return rpm_motor_a;
        case MOTOR_B:  return rpm_motor_b;
        default:       return 0.0f;
    }
}

int32_t Encoder_GetDelta(Motor_ID_t motor)
{
    switch (motor)
    {
        case MOTOR_A:  return encoder_delta_a;
        case MOTOR_B:  return encoder_delta_b;
        default:       return 0;
    }
}

void Encoder_Reset(Motor_ID_t motor)
{
    switch (motor)
    {
        case MOTOR_A:
            __HAL_TIM_SET_COUNTER(&htim2, 0);
            last_count_a = 0;
            break;
        case MOTOR_B:
            __HAL_TIM_SET_COUNTER(&htim3, 0);
            last_count_b = 0;
            break;
        default:
            break;
    }
}



void HAL_TIM_PeriodElapsedCallback(TIM_HandleTypeDef *htim)
{
    if (htim->Instance == TIM6)
    {
        int32_t current_a = (int32_t)__HAL_TIM_GET_COUNTER(&htim2);
        encoder_delta_a   = current_a - last_count_a;
        last_count_a      = current_a;

        rpm_motor_a = (float)encoder_delta_a * 6000.0f / (float)COUNTS_PER_REV;

        int16_t raw_b     = (int16_t)__HAL_TIM_GET_COUNTER(&htim3);
        int16_t current_b = raw_b;
        encoder_delta_b   = (int32_t)-(current_b - (int16_t)last_count_b);
        last_count_b      = (int32_t)current_b;

        rpm_motor_b = (float)encoder_delta_b * 6000.0f / (float)COUNTS_PER_REV;
    }
}
