#ifndef KALMAN_H
#define KALMAN_H

#ifdef __cplusplus
extern "C" {
#endif

typedef struct {
    float angle;
    float bias;

    float P[2][2];

    float Q_angle;
    float Q_bias;
    float R_measure;
} Kalman_t;

void Kalman_SetAngle(Kalman_t *kf, float angle);

float Kalman_Update(Kalman_t *kf, float ref_angle, float gyro_rate, float dt);

#ifdef __cplusplus
}
#endif

#endif
